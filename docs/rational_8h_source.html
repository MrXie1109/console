62 static T gcd(T a, T b) {
63 a = std::abs(a);
64 b = std::abs(b);

66 T temp = b;
67 b = a % b;
68 a = temp;

77 void reduce() {
78 if (den == 0) {
79 throw DomainError("Denominator cannot be zero");
80 }
81 if (den < 0) {
82 num = -num;
83 den = -den;
84 }
85 T g = gcd(num, den);
86 num /= g;
87 den /= g;
88 }
