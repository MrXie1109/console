64 static T gcd(T a, T b) {
65 a = std::abs(a);
66 b = std::abs(b);
67 while (b != 0) {
68 T temp = b;
69 b = a % b;
70 a = temp;
71 }
72 return a;
73 }

79 void reduce() {
80 if (den == 0) {
81 throw DomainError("Denominator cannot be zero");
82 }
83 if (den < 0) {
84 num = -num;
85 den = -den;
86 }
87 T g = gcd(num, den);
88 num /= g;
89 den /= g;
90 }

334 is >> slash >> d;
