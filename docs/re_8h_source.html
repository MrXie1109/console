74 class Match {
75 private:
76 string_type text_copy;
77 struct GroupInfo {
78 std::ptrdiff_t pos;
79 std::ptrdiff_t len;
80 };
81 std::vector<GroupInfo> groups_;
82 bool success;

84 public:
85 Match() : success(false) {}
86 Match(string_type t, const match_type &m) :
87 text_copy(std::move(t)), success(true) {
88 for (size_t i = 0; i < m.size(); ++i)
89 groups_.push_back({m.position(i), m.length(i)});
90 }
91
92 string_type group(int n = 0) const {
93 if (!success || n < 0 || n >= int(groups_.size()))
94 return string_type{};
95 return text_copy.substr(groups_[n].pos, groups_[n].len);
96 }
97
98 std::vector<string_type> groups() const {
99 std::vector<string_type> result;
100 if (!success) return result;
101 for (size_t i = 1; i < groups_.size(); ++i)
102 result.push_back(
103 text_copy.substr(groups_[i].pos, groups_[i].len));
104 return result;
105 }
106
107 int start(int n = 0) const {
108 if (!success || n < 0 || n >= int(groups_.size())) return -1;
109 return int(groups_[n].pos);
110 }
111
112 int end(int n = 0) const {
113 if (!success || n < 0 || n >= int(groups_.size())) return -1;
114 return int(groups_[n].pos + groups_[n].len);
115 }
116
117 std::pair<int, int> span(int n = 0) const {
118 return {start(n), end(n)};
119 }
120
121 explicit operator bool() const { return success; }
122 };

152 std::vector<string_type> result;

163 std::vector<string_type>
164 split(const string_type &text, int maxsplit = 0) const {
165 std::vector<string_type> result;
166 token_iterator it(text.begin(), text.end(), pattern, -1);
167 token_iterator end;
168
169 int count = 0;
170 for (; it != end && (maxsplit <= 0 || count < maxsplit);
171 ++it, ++count)
172 result.push_back(*it);
173
174 if (maxsplit > 0 && count == maxsplit && it != end) {
175 string_type remaining;
176 for (; it != end; ++it) {
177 if (!remaining.empty())
178 remaining += *it;
179 else
180 remaining = *it;
181 }
182 result.push_back(remaining);
183 }
184
185 return result;
186 }

191 string_type sub(const string_type &repl,
192 const string_type &text,
193 int count = 0) const {
194 if (count <= 0) return std::regex_replace(text, pattern, repl);
195
196 string_type result;
197 auto begin = iterator_type(text.begin(), text.end(), pattern);
198 auto end = iterator_type();
199 auto last = text.begin();
200 int replaced = 0;
201
202 for (auto it = begin; it != end && replaced < count;
203 ++it, ++replaced) {
204 result.append(last, text.begin() + it->position());
205 result.append(repl);
206 last = text.begin() + it->position() + it->length();
207 }
208 result.append(last, text.end());
209
210 return result;
211 }
