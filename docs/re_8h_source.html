233 std::vector<std::string>
234 split(const std::string &text, int maxsplit = 0) const {
235 std::vector<std::string> result;
236 std::sregex_token_iterator it(
237 text.begin(), text.end(), pattern, -1);
238 std::sregex_token_iterator end;
239
240 int count = 0;
241 for (; it != end && (maxsplit <= 0 || count < maxsplit);
242 ++it, ++count) {
243 result.push_back(*it);
244 }
245
246 if (maxsplit > 0 && count == maxsplit && it != end) {
247 std::string remaining;
248 for (; it != end; ++it) {
249 if (!remaining.empty())
250 remaining += *it;
251 else
252 remaining = *it;
253 }
254 result.push_back(remaining);
255 }
256
257 return result;
258 }
