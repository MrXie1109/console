75 template <class T = std::string, class CharT = char>
76 T input(const CharT *prompt = "",
77 const BasicInputSettings<CharT> &is = input_settings) {
78 T tmp;
79 while (true) {
80 is.os << prompt << std::flush;
81 is.is >> tmp;
82 if (!is.is) {
83 is.is.clear();
84 is.is.ignore(
85 std::numeric_limits<std::streamsize>::max(), CharT('\n'));
86 is.os << "Stream Error!" << std::endl;
87 continue;
88 }
89 is.is.ignore(
90 std::numeric_limits<std::streamsize>::max(), CharT('\n'));
91 return tmp;
92 }
93 }

223 inline bool inputYesOrNo(const char *prompt = "Type yes or no: ",
224 const InputSettings &is = input_settings) {
225 while (true) {
226 char tmp = inputChar(prompt, is);
227 if (tmp == 'Y' || tmp == 'y')
228 return true;
