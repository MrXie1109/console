76 reference operator*() {
77 if (done()) throw StopIteration("StopIteration");

85 iterator &operator++() {
86 if (done()) return *this;
87 d.advance();
88 return *this;

90
95 const iterator operator++(int) {
96 if (done()) return *this;
97 auto old = *this;
98 d.advance();
99 return old;

107
112 bool operator!=(const iterator &) { return !d.done(); }

120 if (done()) throw StopIteration("StopIteration");
121 auto current = d.current();
122 d.advance();
123 return current;
124 }

131 };
