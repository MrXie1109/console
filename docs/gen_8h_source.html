64 using iterator_category = std::input_iterator_tag;
65
70 iterator(const Derived &d) : d(d) {}
71
77 reference operator*()
78 {
79 if (done())
80 throw StopIteration("StopIteration");
81 return d.current();

83
88 iterator &operator++()
89 {
90 if (done())
91 return *this;
92 d.advance();

101 {
102 if (done())
103 return *this;

105 d.advance();
106 return old;

114
119 bool operator!=(const iterator &) { return !d.done(); }

127 {
128 if (done())
129 throw StopIteration("StopIteration");

131 d.advance();

133 }
134

140 {
141 return d.done();
142 }
