50 {
51 public:
55 class iterator
56 {
57 Derived d;
58
59 public:
60 using value_type = T;
61 using reference = T;
62 using pointer = void;

64 using iterator_category = std::input_iterator_tag;
65
70 iterator(const Derived &d) : d(d) {}
71
77 reference operator*()
78 {
79 if (done())
80 throw StopIteration("StopIteration");
81 return d.current();

83
88 iterator &operator++()
89 {
90 if (done())
91 return *this;
92 d.advance();
93 return *this;
94 }
95
100 const iterator operator++(int)
101 {
102 if (done())
103 return *this;
104 auto old = *this;
105 d.advance();
106 return old;
107 }
108
113 bool operator==(const iterator &) const { return d.done(); }
114
119 bool operator!=(const iterator &) const { return !d.done(); }
120
126 T next()
127 {
128 if (done())
129 throw StopIteration("StopIteration");
130 auto current = d.current();
131 d.advance();
132 return current;
133 }
134
139 bool done()
140 {
141 return d.done();
142 }
143 };
