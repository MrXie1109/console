83 template <class T> struct Task : public TaskBase {
84 std::packaged_task<T()> task;
85 Task(std::packaged_task<T()> &&t) : task(std::move(t)) {}
86 void execute() override { task(); }
87 };

89 std::vector<std::thread> workers;
90 std::queue<std::unique_ptr<TaskBase>> tasks;
91 mutable std::mutex mutex;
92 std::condition_variable cv;
93 std::atomic<bool> shutdown;
94 std::atomic<size_t> active_tasks;
95 launch exit_launch_;
96
97 public:
105 ThreadPool(launch exit_launch = launch::close,
106 size_t num_threads = std::thread::hardware_concurrency()) :
107 shutdown(false), active_tasks(0), exit_launch_(exit_launch) {
108 if (num_threads == 0) num_threads = 2;
109 for (size_t i = 0; i < num_threads; ++i) {
110 workers.emplace_back([this] {
111 while (true) {
112 std::unique_ptr<TaskBase> task;
113 {
114 std::unique_lock<std::mutex> lock(mutex);
115 cv.wait(lock, [this] {
116 return shutdown.load(std::memory_order_acquire)
117 || !tasks.empty();
118 });
119 if (shutdown.load(std::memory_order_acquire)
120 && tasks.empty())
121 return;
122 task = std::move(tasks.front());
123 tasks.pop();
124 }
125 active_tasks.fetch_add(1, std::memory_order_release);
126 try {
127 task->execute();
128 active_tasks.fetch_sub(
129 1, std::memory_order_release);
130 if (active_tasks.load(std::memory_order_acquire)
131 == 0
132 && tasks.empty())
133 cv.notify_all();
134 } catch (...) {
135 active_tasks.fetch_sub(
136 1, std::memory_order_release);
137 if (active_tasks.load(std::memory_order_acquire)
138 == 0
139 && tasks.empty())
140 cv.notify_all();
141 throw;
142 }
143 }
144 });
145 }
146 }

184 template <class F, class... Args>
185 auto
186 submit(F &&f, Args &&...args) -> std::future<decltype(std::forward<F>(
187 f)(std::forward<Args>(args)...))> {
188 using return_type
189 = decltype(std::forward<F>(f)(std::forward<Args>(args)...));
190 auto bound_task
191 = std::bind(std::forward<F>(f), std::forward<Args>(args)...);
192 std::packaged_task<return_type()> task(bound_task);
193 std::future<return_type> result = task.get_future();
194 {
195 std::lock_guard<std::mutex> lock(mutex);
196 if (shutdown.load(std::memory_order_acquire))
197 throw ThreadPoolError("ThreadPool is shutting down");
198 tasks.emplace(std::unique_ptr<TaskBase>(
199 new Task<return_type>(std::move(task))));
200 }
201 cv.notify_one();
202 return result;
203 }

218 template <class F, class Container>
219 auto map(F &&func, const Container &items)
220 -> std::vector<std::future<decltype(func(
221 std::declval<typename Container::value_type>()))>> {
222 using return_type = decltype(func(
223 std::declval<typename Container::value_type>()));
224 std::vector<std::future<return_type>> futures;
225 futures.reserve(items.size());
226 for (const auto &item : items)
227 futures.push_back(submit(func, item));
228 return futures;
229 }

236 void wait() {
237 std::unique_lock<std::mutex> lock(mutex);
238 cv.wait(lock, [this] {
239 return tasks.empty()
240 && active_tasks.load(std::memory_order_acquire) == 0;
241 });
242 }

250 size_t size() const {
251 std::lock_guard<std::mutex> lock(mutex);
252 return tasks.size();
253 }

260 size_t count() const {
261 return active_tasks.load(std::memory_order_acquire);
262 }

270 void close() {
271 shutdown.store(true, std::memory_order_release);
272 cv.notify_all();
273 wait();
274 for (auto &worker : workers)
275 if (worker.joinable()) worker.join();
276 }
277
284 void abort() {
285 shutdown.store(true, std::memory_order_release);
286 {
287 std::lock_guard<std::mutex> lock(mutex);
288 std::queue<std::unique_ptr<TaskBase>>().swap(tasks);
289 }
290 cv.notify_all();
291 for (auto &worker : workers)
292 if (worker.joinable()) worker.detach();
293 }
