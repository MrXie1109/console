75 struct TaskBase
76 {
77 virtual ~TaskBase() = default;
78 virtual void execute() = 0;
79 };
80
87 template <class T>
88 struct Task : public TaskBase
89 {
90 std::packaged_task<T()> task;
91 Task(std::packaged_task<T()> &&t) : task(std::move(t)) {}
92 void execute() override { task(); }
93 };
94
95 std::vector<std::thread> workers;
96 std::queue<std::unique_ptr<TaskBase>> tasks;
97 mutable std::mutex mutex;
98 std::condition_variable cv;
99 std::atomic<bool> shutdown;
100 std::atomic<size_t> active_tasks;
101
102 public:
109 ThreadPoolExecutor(size_t num_threads)
110 : shutdown(false), active_tasks(0)
111 {
112 for (size_t i = 0; i < num_threads; ++i)
113 {
114 workers.emplace_back([this]
115 {
116 while (true) {
117 std::unique_ptr<TaskBase> task;
118 {
119 std::unique_lock<std::mutex> lock(mutex);
120 cv.wait(lock, [this] {
121 return shutdown.load(std::memory_order_acquire) || !tasks.empty();
122 });
123 if (shutdown.load(std::memory_order_acquire) && tasks.empty())
124 return;
125 task = std::move(tasks.front());
126 tasks.pop();
127 }
128 active_tasks.fetch_add(1, std::memory_order_release);
129 task->execute();
130 active_tasks.fetch_sub(1, std::memory_order_release);
131 if (active_tasks.load(std::memory_order_acquire) == 0 && tasks.empty()) {
132 cv.notify_all();
133 }
134 } });
135 }
136 }

158 template <class F, class... Args>
159 auto submit(F &&f, Args &&...args) -> std::future<decltype(f(args...))>
160 {
161 using return_type = decltype(f(args...));
162 auto bound_task = std::bind(std::forward<F>(f), std::forward<Args>(args)...);
163 std::packaged_task<return_type()> task(bound_task);
164 std::future<return_type> result = task.get_future();
165 {
166 std::lock_guard<std::mutex> lock(mutex);
167 if (shutdown.load(std::memory_order_acquire))
168 throw ThreadPoolError("ThreadPoolExecutor is shutting down");
169 tasks.emplace(std::unique_ptr<TaskBase>(new Task<return_type>(std::move(task))));
170 }
171 cv.notify_one();
172 return result;
173 }

186 template <class F, class Container>
187 auto map(F &&func, const Container &items)
188 -> std::vector<std::future<result_type_args<F, typename Container::value_type>>>
189 {
190 using return_type = result_type_args<F, typename Container::value_type>;
191 std::vector<std::future<return_type>> futures;
192 futures.reserve(items.size());
193 for (const auto &item : items)
194 futures.push_back(submit(func, item));
195 return futures;
196 }

203 void wait()
204 {
205 std::unique_lock<std::mutex> lock(mutex);
206 cv.wait(lock, [this]
207 { return tasks.empty() && active_tasks.load(std::memory_order_acquire) == 0; });
208 }

236 void close()
237 {
238 shutdown.store(true, std::memory_order_release);
239 cv.notify_all();
240 wait();
241 for (auto &worker : workers)
242 if (worker.joinable())
243 worker.join();
244 }
