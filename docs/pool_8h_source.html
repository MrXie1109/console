83 template <class T>
84 struct Task : public TaskBase {
85 std::packaged_task<T()> task;

106 ThreadPool(launch exit_launch = launch::close,
107 size_t num_threads = std::thread::hardware_concurrency()) :
108 shutdown(false), active_tasks(0), exit_launch_(exit_launch) {
109 if (num_threads == 0) num_threads = 2;
110 for (size_t i = 0; i < num_threads; ++i) {
111 workers.emplace_back([this] {
112 while (true) {
113 std::unique_ptr<TaskBase> task;
114 {
115 std::unique_lock<std::mutex> lock(mutex);
116 cv.wait(lock, [this] {
117 return shutdown.load(std::memory_order_acquire)
118 || !tasks.empty();
119 });
120 if (shutdown.load(std::memory_order_acquire)
121 && tasks.empty())
122 return;
123 task = std::move(tasks.front());
124 tasks.pop();
125 }
126 active_tasks.fetch_add(1, std::memory_order_release);
127 try {
128 task->execute();
129 active_tasks.fetch_sub(
130 1, std::memory_order_release);
131 if (active_tasks.load(std::memory_order_acquire)
132 == 0
133 && tasks.empty())
134 cv.notify_all();
135 } catch (...) {
136 active_tasks.fetch_sub(
137 1, std::memory_order_release);
138 if (active_tasks.load(std::memory_order_acquire)
139 == 0
140 && tasks.empty())
141 cv.notify_all();
142 throw;
143 }
144 }
145 });
146 }
147 }

185 template <class F, class... Args>
186 auto
187 submit(F &&f, Args &&...args) -> std::future<decltype(std::forward<F>(
188 f)(std::forward<Args>(args)...))> {
189 using return_type
190 = decltype(std::forward<F>(f)(std::forward<Args>(args)...));
191 auto bound_task
192 = std::bind(std::forward<F>(f), std::forward<Args>(args)...);
193 std::packaged_task<return_type()> task(bound_task);
194 std::future<return_type> result = task.get_future();
195 {
196 std::lock_guard<std::mutex> lock(mutex);
197 if (shutdown.load(std::memory_order_acquire))
198 throw ThreadPoolError("ThreadPool is shutting down");
199 tasks.emplace(std::unique_ptr<TaskBase>(
200 new Task<return_type>(std::move(task))));
201 }
202 cv.notify_one();
203 return result;
204 }

219 template <class F, class Container>
220 auto map(F &&func, const Container &items)
221 -> std::vector<std::future<decltype(func(
222 std::declval<typename Container::value_type>()))>> {
223 using return_type = decltype(func(
224 std::declval<typename Container::value_type>()));
225 std::vector<std::future<return_type>> futures;
226 futures.reserve(items.size());
227 for (const auto &item : items)
228 futures.push_back(submit(func, item));
229 return futures;
230 }

237 void wait() {
238 std::unique_lock<std::mutex> lock(mutex);
239 cv.wait(lock, [this] {
240 return tasks.empty()
241 && active_tasks.load(std::memory_order_acquire) == 0;
242 });
243 }

271 void close() {
272 shutdown.store(true, std::memory_order_release);
273 cv.notify_all();
274 wait();
275 for (auto &worker : workers)
276 if (worker.joinable()) worker.join();
277 }

285 void abort() {
286 shutdown.store(true, std::memory_order_release);
287 {
288 std::lock_guard<std::mutex> lock(mutex);
289 std::queue<std::unique_ptr<TaskBase>>().swap(tasks);
290 }
291 cv.notify_all();
292 for (auto &worker : workers)
293 if (worker.joinable()) worker.detach();
294 }
