83 template <class T>
84 struct Task : public TaskBase
85 {
86 std::packaged_task<T()> task;
87 Task(std::packaged_task<T()> &&t) : task(std::move(t)) {}
88 void execute() override { task(); }
89 };
90
91 std::vector<std::thread> workers;
92 std::queue<std::unique_ptr<TaskBase>> tasks;
93 mutable std::mutex mutex;
94 std::condition_variable cv;
95 std::atomic<bool> shutdown;
96 std::atomic<size_t> active_tasks;
97 launch exit_launch_;
98
99 public:
106 ThreadPool(launch exit_launch = launch::close,
107 size_t num_threads = std::thread::hardware_concurrency())
108 : shutdown(false), active_tasks(0), exit_launch_(exit_launch)
109 {
110 if (num_threads == 0)
111 num_threads = 2;
112 for (size_t i = 0; i < num_threads; ++i)
113 {
114 workers.emplace_back([this]
115 {
116 while (true) {
117 std::unique_ptr<TaskBase> task;
118 {
119 std::unique_lock<std::mutex> lock(mutex);
120 cv.wait(lock, [this] {
121 return shutdown.load(std::memory_order_acquire) || !tasks.empty();
122 });
123 if (shutdown.load(std::memory_order_acquire) && tasks.empty())
124 return;
125 task = std::move(tasks.front());
126 tasks.pop();
127 }
128 active_tasks.fetch_add(1, std::memory_order_release);
129 try
130 {
131 task->execute();
132 active_tasks.fetch_sub(1, std::memory_order_release);
133 if (active_tasks.load(std::memory_order_acquire) == 0 && tasks.empty())
134 cv.notify_all();
135 }
136 catch(...)
137 {
138 active_tasks.fetch_sub(1, std::memory_order_release);
139 if (active_tasks.load(std::memory_order_acquire) == 0 && tasks.empty())
140 cv.notify_all();
141 throw;
142 }
143 } });
144 }
145 }

182 template <class F, class... Args>
183 auto submit(F &&f, Args &&...args)
184 -> std::future<decltype(std::forward<F>(f)(std::forward<Args>(args)...))>
185 {
186 using return_type = decltype(std::forward<F>(f)(std::forward<Args>(args)...));
187 auto bound_task = std::bind(std::forward<F>(f), std::forward<Args>(args)...);
188 std::packaged_task<return_type()> task(bound_task);
189 std::future<return_type> result = task.get_future();
190 {
191 std::lock_guard<std::mutex> lock(mutex);
192 if (shutdown.load(std::memory_order_acquire))
193 throw ThreadPoolError("ThreadPool is shutting down");
194 tasks.emplace(std::unique_ptr<TaskBase>(new Task<return_type>(std::move(task))));
195 }
196 cv.notify_one();
197 return result;
198 }

212 template <class F, class Container>
213 auto map(F &&func, const Container &items)
214 -> std::vector<std::future<decltype(func(std::declval<typename Container::value_type>()))>>
215 {
216 using return_type = decltype(func(std::declval<typename Container::value_type>()));
217 std::vector<std::future<return_type>> futures;
218 futures.reserve(items.size());
219 for (const auto &item : items)
220 futures.push_back(submit(func, item));
221 return futures;
222 }

229 void wait()
230 {
231 std::unique_lock<std::mutex> lock(mutex);
232 cv.wait(lock, [this]
233 { return tasks.empty() && active_tasks.load(std::memory_order_acquire) == 0; });
234 }

262 void close()
263 {
264 shutdown.store(true, std::memory_order_release);
265 cv.notify_all();
266 wait();
267 for (auto &worker : workers)
268 if (worker.joinable())
269 worker.join();
270 }
271
277 void abort()
278 {
279 shutdown.store(true, std::memory_order_release);
280 {
281 std::lock_guard<std::mutex> lock(mutex);
282 std::queue<std::unique_ptr<TaskBase>>().swap(tasks);
283 }
284 cv.notify_all();
285 for (auto &worker : workers)
286 if (worker.joinable())
287 worker.detach();
288 }
