171 template <class Gen>
172 class Window : public Generator<Window<Gen>,
173 std::vector<typename Gen::value_type>> {
174 Gen gen_;
175 size_t window_size_;
176 std::vector<typename Gen::value_type> window_;
177 bool started = false;
178
179 void init() {
180 if (!started) {
181 for (size_t i = 0; i < window_size_ && !gen_.done(); ++i) {
182 window_.push_back(gen_.current());
183 gen_.advance();
184 }
185 started = true;
186 }
187 }
188
189 public:
196 explicit Window(Gen gen, size_t window_size = 16) :
197 gen_(gen), window_size_(window_size) {
198 if (window_size_ == 0)
199 throw ValueError("Window size must be greater than 0");
200 }
201
206 bool done() {
207 init();
208 return window_.size() < window_size_;
209 }
210
215 std::vector<typename Gen::value_type> current() {
216 init();
217 return window_;
218 }
219
223 void advance() {
224 init();
225 if (!gen_.done()) {
226 window_.erase(window_.begin());
227 window_.push_back(gen_.current());
228 gen_.advance();
229 } else
230 window_.erase(window_.begin());
231 }
232 };

498 template <class Gen, class Pred>
499 class DropUntil
500 : public Generator<DropUntil<Gen, Pred>, typename Gen::value_type> {
501 Gen gen;
502 Pred pred;
503 bool dropped = false;
504
505 void do_drop() {
506 while (!gen.done() && !pred(gen.current())) gen.advance();
507 if (!gen.done() && pred(gen.current())) gen.advance();
508 dropped = true;
509 }
510
511 public:
517 DropUntil(Gen g, Pred p) : gen(g), pred(p) {}
518
523 bool done() {
524 if (!dropped) do_drop();
525 return gen.done();
526 }
527
532 auto current() -> decltype(gen.current()) { return gen.current(); }

537 void advance() { gen.advance(); }
