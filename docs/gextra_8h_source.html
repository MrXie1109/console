176 template <class Gen>
177 class Window : public Generator<Window<Gen>,
178 std::vector<typename Gen::value_type>> {
179 Gen gen_;
180 size_t window_size_;
181 std::vector<typename Gen::value_type> window_;
182 bool started = false;
183
184 void init() {
185 if (!started) {
186 for (size_t i = 0; i < window_size_ && !gen_.done(); ++i) {
187 window_.push_back(gen_.current());
188 gen_.advance();
189 }
190 started = true;
191 }
192 }
193
194 public:
201 explicit Window(Gen gen, size_t window_size = 16) :
202 gen_(gen), window_size_(window_size) {
203 if (window_size_ == 0)
204 throw ValueError("Window size must be greater than 0");
205 }
206
211 bool done() {
212 init();
213 return window_.size() < window_size_;
214 }
215
220 std::vector<typename Gen::value_type> current() {
221 init();
222 return window_;
223 }
224
228 void advance() {
229 init();
230 if (!gen_.done()) {
231 window_.erase(window_.begin());
232 window_.push_back(gen_.current());
233 gen_.advance();
234 } else
235 window_.erase(window_.begin());
236 }
237 };
