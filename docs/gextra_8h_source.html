169 template <class Gen>
170 class Window : public Generator<Window<Gen>,
171 std::vector<typename Gen::value_type>> {
172 Gen gen_;
173 size_t window_size_;
174 std::vector<typename Gen::value_type> window_;
175 bool started = false;
176
177 void init() {
178 if (!started) {
179 for (size_t i = 0; i < window_size_ && !gen_.done(); ++i) {
180 window_.push_back(gen_.current());
181 gen_.advance();
182 }
183 started = true;
184 }
185 }
186
187 public:
194 explicit Window(Gen gen, size_t window_size = 16) :
195 gen_(gen), window_size_(window_size) {
196 if (window_size_ == 0)
197 throw ValueError("Window size must be greater than 0");
198 }
199
204 bool done() {
205 init();
206 return window_.size() < window_size_;
207 }
208
213 std::vector<typename Gen::value_type> current() {
214 init();
215 return window_;
216 }
217
221 void advance() {
222 init();
223 if (!gen_.done()) {
224 window_.erase(window_.begin());
225 window_.push_back(gen_.current());
226 gen_.advance();
227 } else
228 window_.erase(window_.begin());
229 }
230 };
