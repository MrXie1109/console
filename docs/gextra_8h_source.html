173 template <class Gen>
174 class Window : public Generator<Window<Gen>,
175 std::vector<typename Gen::value_type>> {
176 Gen gen_;
177 size_t window_size_;
178 std::vector<typename Gen::value_type> window_;
179 bool started = false;
180
181 void init() {
182 if (!started) {
183 for (size_t i = 0; i < window_size_ && !gen_.done(); ++i) {
184 window_.push_back(gen_.current());
185 gen_.advance();
186 }
187 started = true;
188 }
189 }
190
191 public:
198 explicit Window(Gen gen, size_t window_size = 16) :
199 gen_(gen), window_size_(window_size) {
200 if (window_size_ == 0)
201 throw ValueError("Window size must be greater than 0");
202 }
203
208 bool done() {
209 init();
210 return window_.size() < window_size_;
211 }
212
217 std::vector<typename Gen::value_type> current() {
218 init();
219 return window_;
220 }
221
225 void advance() {
226 init();
227 if (!gen_.done()) {
228 window_.erase(window_.begin());
229 window_.push_back(gen_.current());
230 gen_.advance();
231 } else
232 window_.erase(window_.begin());
233 }
234 };
