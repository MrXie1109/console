186 template <class C>
187 void shuffle(C &&c, std::mt19937 &gen = default_gen())
188 {
189 if (std::begin(c) == std::end(c))
190 return;
191 for (size_t i = c.size() - 1; i > 0; i--)
192 {
193 auto j = randint<size_t>(0, i, gen);
194 std::swap(*std::next(std::begin(c), i),
195 *std::next(std::begin(c), j));
196 }
197 }

451 template <class C>
452 std::vector<typename C::value_type>
453 sample(const C &c, size_t size, bool replace = false,
454 std::mt19937 &gen = default_gen())
455 {
456 using value_type = typename C::value_type;
457 std::vector<value_type> result;
458 result.reserve(size);
459 auto it_begin = std::begin(c);
460 auto it_end = std::end(c);
461 size_t n = std::distance(it_begin, it_end);
462 if (n == 0)
463 throw ContainerError(
464 "Cannot sample from empty container");
465 if (!replace && size > n)
466 throw ContainerError(
467 "Sample size exceeds container size when replace=false");
468 if (replace)
469 for (size_t i = 0; i < size; ++i)
470 {
471 result.push_back(*std::next(
472 it_begin,
473 std::uniform_int_distribution<size_t>(0, n - 1)(gen)));
474 }
475 else
476 {
477 std::vector<size_t> indices(n);
478 std::iota(indices.begin(), indices.end(), 0);
479 shuffle(indices, gen);
480 for (size_t i = 0; i < size; ++i)
481 result.push_back(*std::next(it_begin, indices[i]));
482 }
483 return result;
484 }
