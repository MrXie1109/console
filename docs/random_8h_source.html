175 template <class C> void shuffle(C &&c, std::mt19937 &gen = default_gen()) {
176 if (std::begin(c) == std::end(c)) return;
177 for (size_t i = c.size() - 1; i > 0; i--) {
178 auto j = randint<size_t>(0, i, gen);
179 std::swap(
180 *std::next(std::begin(c), i), *std::next(std::begin(c), j));
181 }
182 }

415 template <class C>
416 std::vector<typename C::value_type> sample(const C &c,
417 size_t size,
418 bool replace = false,
419 std::mt19937 &gen = default_gen()) {
420 using value_type = typename C::value_type;
421 std::vector<value_type> result;
422 result.reserve(size);
423 auto it_begin = std::begin(c);
424 auto it_end = std::end(c);
425 size_t n = std::distance(it_begin, it_end);
426 if (n == 0) throw ContainerError("Cannot sample from empty container");
427 if (!replace && size > n)
428 throw ContainerError(
429 "Sample size exceeds container size when replace=false");
430 if (replace)
431 for (size_t i = 0; i < size; ++i) {
432 result.push_back(*std::next(it_begin,
433 std::uniform_int_distribution<size_t>(0, n - 1)(gen)));
434 }
435 else {
436 std::vector<size_t> indices(n);
437 std::iota(indices.begin(), indices.end(), 0);
438 shuffle(indices, gen);
439 for (size_t i = 0; i < size; ++i)
440 result.push_back(*std::next(it_begin, indices[i]));
441 }
442 return result;
443 }
