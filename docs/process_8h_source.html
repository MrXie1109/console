272 ExitStatus wait() const

393 ProcessInfo get_info() const
394 {
395 ProcessInfo info;
396 info.pid = pid_;
397 info.running = is_alive();
398 char path[64];
399 snprintf(path, sizeof(path), "/proc/%d/stat", pid_);
400 FILE *fp = fopen(path, "r");
401 if (fp)
402 {
403 unsigned long utime, stime;
404 int ppid;
405 fscanf(fp, "%d %*s %*c %d %*d %*d %*d %*d %*d %*d %*d %*d %*d %lu %lu",
406 &info.pid, &ppid, &utime, &stime);
407 info.ppid = ppid;
408 info.cpu_time = (utime + stime) / (double)sysconf(_SC_CLK_TCK);
409 fclose(fp);

412 fp = fopen(path, "r");
413 if (fp)
414 {
