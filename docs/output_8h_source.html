58 inline std::basic_ostream<CharT, Traits> &cont_print_sequence(
59 std::basic_ostream<CharT, Traits> &os, const Cont &cont) {
60 if (begin(cont) == end(cont)) return os << CharT('[') << CharT(']');
61 auto it = begin(cont);
62 os << CharT('[');
63 repr(*it, os);
64 while (++it != end(cont)) {
65 os << CharT(','), os << CharT(' ');
66 repr(*it, os);
67 }
68 return os << CharT(']');
69 }

386 class BasicOutput {
387 using string_type = std::basic_string<CharT, Traits>;
388
389 std::basic_ostream<CharT, Traits> &os;
390 string_type sep;
391 string_type end;
392 bool isFlush;
393
394 public:
398 BasicOutput() :
399 os(std::cout), sep(string_type{1, CharT(' ')}),
400 end(string_type{1, CharT('\n')}), isFlush(true) {}
401
409 BasicOutput(std::basic_ostream<CharT, Traits> &o,
410 const string_type &s,
411 const string_type &e,
412 bool isF) : os(o), sep(s), end(e), isFlush(isF) {}
413
418 BasicOutput &operator()() {
419 os << end;
420 if (isFlush) os << std::flush;
421 return *this;
422 }

430 template <class T>

445 template <class T, class... Args>
446 BasicOutput &operator()(const T &t, const Args &...args) {
447 put(os, t);
448 os << sep;
