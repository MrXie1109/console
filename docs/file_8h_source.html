198 void write_lines(const std::vector<std::string> &lines) const {
199 std::ofstream fout(path, std::ios::binary);
200 if (lines.empty()) return;
201 if (!fout.is_open())
202 throw FileError("Cannot Open File \"" + path + '"');
203 auto it = lines.begin();
204 fout << *it;
205 while (++it != lines.end()) fout << '\n' << *it;
206 if (!fout.good())
207 throw FileError("The Stream of \"" + path + "\" is Not Good");
208 }
