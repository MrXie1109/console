106 string_type read_text() const {
107 std::ifstream fin(narrow());
108 if (!fin.is_open())
109 throw FileError("Cannot Open File \"" + narrow() + '"');
110 std::string buf{std::istreambuf_iterator<char>(fin),
111 std::istreambuf_iterator<char>()};
112 if (fin.fail() && !fin.eof())
113 throw FileError(
