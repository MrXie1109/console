290 int readRaw() {
291#ifdef _WIN32
292 if (_kbhit()) return _getch();
293 return -1;
294#else
295 struct pollfd pfd = {STDIN_FILENO, POLLIN, 0};
296 if (poll(&pfd, 1, 0) <= 0) return -1;
297 unsigned char c;
298 if (read(STDIN_FILENO, &c, 1) != 1) return -1;
299 return c;
300#endif
301 }

309 int readRawBlock() {
310#ifdef _WIN32
311 return _getch();
312#else
313 unsigned char c;
314 struct pollfd pfd = {STDIN_FILENO, POLLIN, 0};
315 while (true) {
316 if (poll(&pfd, 1, -1) > 0) {
317 if (read(STDIN_FILENO, &c, 1) == 1) return c;
318 }
319 }
320#endif
321 }

335 Key parse(int raw) {
336 if ((raw >= 'A' && raw <= 'Z') || (raw >= '0' && raw <= '9')) {
337 return static_cast<Key>(raw);
338 }
339 if (raw >= 'a' && raw <= 'z') {
340 return static_cast<Key>(raw - 32);
341 }
342 switch (raw) {
343 case ' ':
344 return Key::Space;
345 case 13:
346 return Key::Enter;
347 case 27:
348 return handleEscape();
349 case 8:
350 case 127:
351 return Key::Backspace;
352 case 9:
353 return Key::Tab;
354 }
355#ifdef _WIN32
356 if (raw == 0xE0 || raw == 0x00) {
357 int ext = _getch();
358 switch (ext) {
359 case 0x48:
360 return Key::Up;
361 case 0x50:
362 return Key::Down;
363 case 0x4B:
364 return Key::Left;
365 case 0x4D:
366 return Key::Right;
367 case 0x3B:
368 return Key::F1;
369 case 0x3C:
370 return Key::F2;
371 case 0x3D:
372 return Key::F3;
373 case 0x3E:
374 return Key::F4;
375 case 0x3F:
376 return Key::F5;
377 case 0x40:
378 return Key::F6;
379 case 0x41:
380 return Key::F7;

383 case 0x43:
384 return Key::F9;

386 return Key::F10;

404 Key handleEscape() {
405#ifndef _WIN32
406 struct pollfd pfd = {STDIN_FILENO, POLLIN, 0};
407 if (poll(&pfd, 1, 5) <= 0) return Key::Esc;
408 char seq[3];
409 int n = read(STDIN_FILENO, seq, 3);
410 if (n >= 2 && seq[0] == '[') {
411 switch (seq[1]) {
412 case 'A':
413 return Key::Up;
414 case 'B':
415 return Key::Down;
416 case 'C':
417 return Key::Right;
418 case 'D':
419 return Key::Left;
420 }
421 }
422#endif
423 return Key::Esc;
424 }
