154 inline std::pair<int, int> get_terminal_size() noexcept {
155#ifdef _WIN32
156 CONSOLE_SCREEN_BUFFER_INFO csbi;
157 HANDLE hConsole = GetStdHandle(STD_OUTPUT_HANDLE);
158 if (GetConsoleScreenBufferInfo(hConsole, &csbi)) {
159 int cols = csbi.srWindow.Right - csbi.srWindow.Left + 1;
160 int rows = csbi.srWindow.Bottom - csbi.srWindow.Top + 1;
161 return {rows, cols};
162 }
163 return {0, 0};
164#elif defined(__unix__) || defined(__APPLE__)
165 struct winsize w;
166 if (ioctl(STDOUT_FILENO, TIOCGWINSZ, &w) == 0)
167 return {w.ws_row, w.ws_col};
168 return {0, 0};
169#else
170 return {0, 0};
171#endif
172 }

248#ifdef _WIN32
253 inline const bool &init_windows_terminal() noexcept {
254 static bool initialized = []() noexcept -> bool {
255 HANDLE hConsole = GetStdHandle(STD_OUTPUT_HANDLE);
256 if (hConsole == INVALID_HANDLE_VALUE || hConsole == nullptr)
257 return false;
258 DWORD mode;
259 if (!GetConsoleMode(hConsole, &mode)) return false;
260 mode |= ENABLE_VIRTUAL_TERMINAL_PROCESSING;
261 mode |= ENABLE_PROCESSED_OUTPUT;
262
263 return SetConsoleMode(hConsole, mode) != 0;
264 }();
265 return initialized;
266 }
