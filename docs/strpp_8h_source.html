157 inline std::string title(std::string str) {
158 if (str.empty()) return "";
159 if (str[0] >= 'a' && str[0] <= 'z') str[0] += 'A' - 'a';
160 for (size_t i = 1; i < str.size(); ++i) {
161 if (isspace((unsigned char)str[i - 1]) && str[i] >= 'a'
162 && str[i] <= 'z')
163 str[i] += 'A' - 'a';
164 else if (str[i] >= 'A' && str[i] <= 'Z')
165 str[i] -= 'A' - 'a';

215 inline std::vector<std::string>
216 split(std::string text, const std::string &sep = " ") {
217 std::vector<std::string> vec;
218 size_t start = 0;
219 size_t end;
220 size_t sep_len = sep.length();
221 while ((end = text.find(sep, start)) != std::string::npos) {
222 vec.emplace_back(text, start, end - start);
223 start = end + sep_len;
224 }
225 vec.emplace_back(text, start);
226 return vec;
227 }
