101 auto front = std::find_if(str.begin(),
102 str.end(),
103 [](unsigned char uc) -> bool { return !isspace(uc); });
104 if (front == str.end()) {
105 str.clear();
106 return str;

235 template <class CharT, class Traits, class Allocator>
236 std::basic_string<CharT, Traits, Allocator>
237 title(std::basic_string<CharT, Traits, Allocator> str,
238 const std::locale &loc = std::locale{}) {
239 using string_type = std::basic_string<CharT, Traits, Allocator>;
240 if (str.empty()) return string_type{};
241 str[0] = std::toupper(str[0], loc);
242 for (size_t i = 1; i < str.size(); ++i) {
243 if (std::isspace(str[i - 1], loc))
244 str[i] = std::toupper(str[i], loc);
245 else
246 str[i] = std::tolower(str[i], loc);
247 }
248 return str;
249 }

297 template <class CharT, class Traits, class Allocator>
298 BasicPartitionResult<CharT, Traits, Allocator>
299 partition(const std::basic_string<CharT, Traits, Allocator> &text,
300 const std::basic_string<CharT, Traits, Allocator> &sep) {
301 using string_type = std::basic_string<CharT, Traits, Allocator>;
302 size_t pos = text.find(sep);
303 if (pos == string_type::npos)
304 return BasicPartitionResult<CharT, Traits, Allocator>{
305 text, string_type{}, string_type{}};
306 return BasicPartitionResult<CharT, Traits, Allocator>{
307 text.substr(0, pos), sep, text.substr(pos + sep.size())};
308 }

320 template <class CharT, class Traits, class Allocator>
321 std::vector<std::basic_string<CharT, Traits, Allocator>>
322 split(const std::basic_string<CharT, Traits, Allocator> &text,
323 const std::basic_string<CharT, Traits, Allocator> &sep
324 = std::basic_string<CharT, Traits, Allocator>{1, CharT{' '}}) {
325 using string_type = std::basic_string<CharT, Traits, Allocator>;
326 std::vector<string_type> vec;
327 size_t sep_len = sep.length();
328 if (sep_len == 0) {
329 vec.push_back(text);
330 return vec;
331 }
332 size_t start = 0;
333 size_t end;
334 while ((end = text.find(sep, start)) != string_type::npos) {
335 vec.emplace_back(text, start, end - start);
336 start = end + sep_len;
337 }
338 vec.emplace_back(text, start);
339 return vec;
340 }
