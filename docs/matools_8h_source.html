64 return double(sum(arr)) / arr.fsize();

76 template <class T, size_t... Dims>
77 double variance(const MultiArray<T, Dims...> &arr, bool sample = true) {
78 double m = mean(arr);
79 double sq_sum = 0;
80 arr.for_each([&](const T &x) {
81 double d = double(x) - m;
82 sq_sum += d * d;
83 });
84 return sq_sum / (arr.fsize() - (sample ? 1 : 0));
85 }

96 template <class T, size_t... Dims>
97 double stddev(const MultiArray<T, Dims...> &arr, bool sample = true) {
98 return std::sqrt(variance(arr, sample));
99 }

148 template <class T, size_t N>
149 MultiArray<T, N> normalize(const MultiArray<T, N> &a) {
150 double len = norm(a);
151 if (len == 0) return a;
152 MultiArray<T, N> result;
153 auto ai = a.fbegin();
154 auto ri = result.fbegin();
155 while (ri != result.fend()) *ri++ = *ai++ / len;
156 return result;
157 }

167 template <class T, size_t N>
168 double euclidean(const MultiArray<T, N> &a, const MultiArray<T, N> &b) {
169 double sum = 0;
170 auto ai = a.fbegin();
171 auto bi = b.fbegin();
172 while (ai != a.fend()) {
173 double d = *ai++ - *bi++;
174 sum += d * d;
175 }
176 return std::sqrt(sum);
177 }

187 template <class T, size_t N>
188 double manhattan(const MultiArray<T, N> &a, const MultiArray<T, N> &b) {
189 double sum = 0;
190 auto ai = a.fbegin();
191 auto bi = b.fbegin();
192 while (ai != a.fend()) sum += std::abs(*ai++ - *bi++);
193 return sum;
194 }

208 template <class T, size_t M, size_t N, size_t K>
209 MultiArray<T, M, K>
210 matmul(const MultiArray<T, M, N> &A, const MultiArray<T, N, K> &B) {
211 MultiArray<T, M, K> C{};
212 for (size_t i = 0; i < M; ++i)
213 for (size_t j = 0; j < N; ++j)
214 for (size_t k = 0; k < K; ++k) C[i][k] += A[i][j] * B[j][k];
215 return C;
216 }

402 MultiArray<T, N + K - 1> result{};
