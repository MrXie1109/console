62 template <class T, size_t... Dims>

64 return double(sum(arr)) / arr.fsize();

76 template <class T, size_t... Dims>
77 double variance(const MultiArray<T, Dims...> &arr, bool sample = true) {
78 double m = mean(arr);
79 double sq_sum = 0;
80 arr.for_each([&](const T &x) {
81 double d = double(x) - m;
82 sq_sum += d * d;
83 });
84 return sq_sum / (arr.fsize() - (sample ? 1 : 0));
85 }

96 template <class T, size_t... Dims>
97 double stddev(const MultiArray<T, Dims...> &arr, bool sample = true) {
98 return std::sqrt(variance(arr, sample));
99 }

147 template <class T, size_t N>
148 MultiArray<T, N> normalize(const MultiArray<T, N> &a) {
149 double len = norm(a);
150 if (len == 0) return a;
151 MultiArray<T, N> result;
152 auto ai = a.fbegin();
153 auto ri = result.fbegin();
154 while (ri != result.fend()) *ri++ = *ai++ / len;
155 return result;
156 }

166 template <class T, size_t N>
167 double euclidean(const MultiArray<T, N> &a, const MultiArray<T, N> &b) {
168 double sum = 0;
169 auto ai = a.fbegin();
170 auto bi = b.fbegin();
171 while (ai != a.fend()) {
172 double d = *ai++ - *bi++;
173 sum += d * d;
174 }
175 return std::sqrt(sum);
176 }

186 template <class T, size_t N>
187 double manhattan(const MultiArray<T, N> &a, const MultiArray<T, N> &b) {
188 double sum = 0;
189 auto ai = a.fbegin();
190 auto bi = b.fbegin();
191 while (ai != a.fend()) sum += std::abs(*ai++ - *bi++);
192 return sum;
193 }

207 template <class T, size_t M, size_t N, size_t K>
208 MultiArray<T, M, K>
209 matmul(const MultiArray<T, M, N> &A, const MultiArray<T, N, K> &B) {
210 MultiArray<T, M, K> C{};
211 for (size_t i = 0; i < M; ++i)
212 for (size_t j = 0; j < N; ++j)
213 for (size_t k = 0; k < K; ++k) C[i][k] += A[i][j] * B[j][k];
214 return C;
215 }

229 for (size_t j = 0; j < N; j++) B[j][i] = A[i][j];
230 return B;

398 convolve1d(const MultiArray<T, N> &signal, const MultiArray<T, K> &kernel) {
399 MultiArray<T, N + K - 1> result{};
400 for (size_t i = 0; i < N; i++)
