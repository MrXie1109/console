68 void point(unsigned x, unsigned y, bool on = true) {
69 if (x >= COLS || y >= ROWS) return;
70 unsigned braille_col = x / 2;
71 unsigned braille_row = y / 4;
72 unsigned sub_x = x % 2;
73 unsigned sub_y = y % 4;
74 static const unsigned char mapping[2][4]
75 = {{0, 1, 2, 6}, {3, 4, 5, 7}};
76 unsigned char bit = 1 << mapping[sub_x][sub_y];
77 if (on)
78 masks[braille_row][braille_col] |= bit;
79 else
80 masks[braille_row][braille_col] &= ~bit;
81 }

92 void line(int x0, int y0, int x1, int y1, bool on = true) {
93 int dx = std::abs(x1 - x0);
94 int dy = -std::abs(y1 - y0);
95 int sx = x0 < x1 ? 1 : -1;
96 int sy = y0 < y1 ? 1 : -1;
97 int err = dx + dy;
98 while (true) {
99 point(x0, y0, on);
100 if (x0 == x1 && y0 == y1) break;
101 int e2 = 2 * err;
102 if (e2 >= dy) err += dy, x0 += sx;
103 if (e2 <= dx) err += dx, y0 += sy;
104 }
105 }

132 void fillRect(int x0, int y0, int x1, int y1, bool on = true) {
133 if (x0 > x1) std::swap(x0, x1);
134 if (y0 > y1) std::swap(y0, y1);
135 for (int y = y0; y <= y1; ++y)

147 void circle(int cx, int cy, int r, bool on = true) {
148 int x = 0, y = r;
149 int d = 3 - 2 * r;
150 while (x <= y) {
151 point(cx + x, cy + y, on);
152 point(cx - x, cy + y, on);
153 point(cx + x, cy - y, on);
154 point(cx - x, cy - y, on);
155 point(cx + y, cy + x, on);
156 point(cx - y, cy + x, on);
157 point(cx + y, cy - x, on);
158 point(cx - y, cy - x, on);
159 if (d < 0)
160 d = d + 4 * x + 6;
161 else
162 d = d + 4 * (x - y) + 10, y--;
163 x++;
164 }

175 void fillCircle(int cx, int cy, int r, bool on = true) {
176 int x = 0, y = r;
177 int d = 3 - 2 * r;
178 while (x <= y) {
179 for (int i = cx - x; i <= cx + x; ++i) {
180 point(i, cy + y, on);
181 point(i, cy - y, on);
182 }
183 for (int i = cx - y; i <= cx + y; ++i) {
184 point(i, cy + x, on);

213 template <class CharT, class Traits = std::char_traits<CharT>>
214 void update(std::basic_ostream<CharT, Traits> &os = std::cout,
215 bool clear = false) const {
216 if (clear) {
217 os << "\033[2J";
218 }
219 std::string out;
220 out.reserve((ROWS / 4) * (COLS / 2 * 3 + 1));
221 for (unsigned row = 0; row < ROWS / 4; ++row) {
222 for (unsigned col = 0; col < COLS / 2; ++col) {
223 unsigned int code = 0x2800 + masks[row][col];
224 out += static_cast<char>(0xE0 | (code >> 12));
225 out += static_cast<char>(0x80 | ((code >> 6) & 0x3F));
226 out += static_cast<char>(0x80 | (code & 0x3F));
227 }
228 out += '\n';
229 }
230 os << "\033[H" << out.c_str() << std::flush;
231 }
