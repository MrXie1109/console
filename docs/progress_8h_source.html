102 template <class CharT, class Traits, class Iter>
103 class BasicProgress {
104 public:
105 using config_type = BasicProgressConfig<CharT, Traits>;
106
111 class iterator {
112 const config_type *config_;
113 size_t current_;
114 size_t total_;
115 Iter it_;
116 std::chrono::steady_clock::time_point last_draw_;
117
118 void draw() {
119 if (!config_) return;
120 auto now = std::chrono::steady_clock::now();
121 if (current_ < total_
122 && (now - last_draw_) < std::chrono::milliseconds{50}) {
123 return;
124 }
125 int percent = current_ * 100 / total_;
126 int filled = percent * config_->width / 100;
127 config_->os << CharT('\r') << config_->prefix;
128 for (int i = 0; i < filled; ++i)
129 config_->os << config_->fill_char;
130 for (int i = filled; i < config_->width; ++i)
131 config_->os << config_->empty_char;
132 config_->os << config_->suffix;
133 if (config_->show_percent) {
134 config_->os.width(4);
135 config_->os << percent << CharT('%');
136 }
137 config_->os.flush();
138 last_draw_ = now;
139 }
140
141 public:
142 iterator(const config_type *config,
143 size_t current,
144 size_t total,
145 Iter it) :
146 config_(config), current_(current), total_(total), it_(it),
147 last_draw_(std::chrono::steady_clock::now()) {}
148
149 iterator &operator++() {
150 if (current_ < total_) {
151 ++current_;
152 ++it_;
153 draw();
154 }
155 return *this;
156 }
157
158 iterator operator++(int) {
159 iterator tmp = *this;
160 ++(*this);
161 return tmp;
162 }
163
164 bool operator==(const iterator &other) {
165 return current_ == other.current_;
166 }
167
168 bool operator!=(const iterator &other) {
169 return current_ != other.current_;
170 }
171
172 decltype(*it_) operator*() { return *it_; }
173 };
