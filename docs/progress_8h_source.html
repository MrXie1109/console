112 void draw() {
113 if (!config_) return;
114 auto now = std::chrono::steady_clock::now();
115 if (current_ < total_
116 && (now - last_draw_) < std::chrono::milliseconds{50}) {
117 return;
118 }
119 int percent = current_ * 100 / total_;
120 int filled = percent * config_->width / 100;
121 config_->os << '\r' << config_->prefix;
122 for (int i = 0; i < filled; ++i)
123 config_->os << config_->fill_char;
124 for (int i = filled; i < config_->width; ++i)
125 config_->os << config_->empty_char;
126 config_->os << config_->suffix;
127 if (config_->show_percent) {
128 config_->os.width(4);
129 config_->os << percent << '%';
130 }
131 config_->os.flush();
132 last_draw_ = now;
133 }
