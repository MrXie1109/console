246 inline double fps(double target) {
247 using namespace std::chrono;
248 thread_local auto last = steady_clock::now(); //于首次调用时初始化
249 std::this_thread::sleep_until(last + duration<double>(1.0 / target));
250 double actual
251 = 1.0 / duration<double>(steady_clock::now() - last).count();
252 last = steady_clock::now();
253 return actual;

358 int iso_week() const {
359 // 取当年 1 月 4 日所在的周一为准
360 DateTime jan4(year(), 1, 4);
361 int dow = jan4.iso_weekday();
362 int week_num = (yearday() - jan4.yearday() + dow + 6) / 7;
363 if (week_num < 1)
364 week_num = DateTime(year() - 1, 12, 31).iso_week();
365 if (week_num > 52 && days_in_month() == 31 && month() == 12
366 && iso_weekday() <= 3)
367 week_num = 1;
368 return week_num;
369 }
