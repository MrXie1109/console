150 inline void transform(uint32_t *state, const uint8_t *data)
151 {
152 uint32_t W[64];
153 uint32_t A, B, C, D, E, F, G, H, T1, T2;
154 for (int i = 0; i < 16; i++)
155 W[i] = (data[i * 4] << 24) | (data[i * 4 + 1] << 16) |
156 (data[i * 4 + 2] << 8) | data[i * 4 + 3];
157 for (int i = 16; i < 64; i++)
158 W[i] = gamma1(W[i - 2]) + W[i - 7] + gamma0(W[i - 15]) + W[i - 16];
159 A = state[0];
160 B = state[1];
161 C = state[2];
162 D = state[3];
163 E = state[4];
164 F = state[5];
165 G = state[6];
166 H = state[7];
167 for (int i = 0; i < 64; i++)
168 {
169 T1 = H + sigma1(E) + ch(E, F, G) + K[i] + W[i];
170 T2 = sigma0(A) + maj(A, B, C);
171 H = G;
172 G = F;
173 F = E;
174 E = D + T1;
175 D = C;
176 C = B;
177 B = A;
178 A = T1 + T2;
179 }
180 state[0] += A;
181 state[1] += B;
182 state[2] += C;
183 state[3] += D;
184 state[4] += E;
185 state[5] += F;
186 state[6] += G;
187 state[7] += H;
188 }

336 inline void transform(uint32_t *state, const uint8_t *block)
337 {
338 uint32_t a = state[0], b = state[1], c = state[2], d = state[3];
339 uint32_t M[16];
340 for (int i = 0; i < 16; i++)
341 {
342 M[i] = block[i * 4] | (block[i * 4 + 1] << 8) |
343 (block[i * 4 + 2] << 16) | (block[i * 4 + 3] << 24);
344 }
345 for (int i = 0; i < 64; i++)
346 {
347 uint32_t f, g;
348 if (i < 16)
349 {
350 f = F(b, c, d);
351 g = i;
352 }
353 else if (i < 32)
354 {
355 f = G(b, c, d);
356 g = (5 * i + 1) % 16;
357 }
358 else if (i < 48)
359 {
360 f = H(b, c, d);
361 g = (3 * i + 5) % 16;
362 }
363 else
364 {
365 f = I(b, c, d);
366 g = (7 * i) % 16;
367 }
368 uint32_t temp = d;
369 d = c;
370 c = b;
371 b = b + rotate_left(a + f + K[i] + M[g], S[i]);
372 a = temp;
373 }
374 state[0] += a;
375 state[1] += b;
376 state[2] += c;
377 state[3] += d;
378 }

414 inline std::string base64_encode(const std::string &input)
415 {
416 const char *base64_chars = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/";
417 std::string result;
418 result.reserve(((input.size() + 2) / 3) * 4);
419 uint32_t buffer = 0;
420 int bits_collected = 0;
421 for (char c : input)
422 {
423 buffer = (buffer << 8) | static_cast<uint8_t>(c);
424 bits_collected += 8;
425 while (bits_collected >= 6)
426 {
427 bits_collected -= 6;
428 result.push_back(base64_chars[(buffer >> bits_collected) & 0x3F]);
429 }
430 }
431 if (bits_collected > 0)
432 {
433 buffer <<= (6 - bits_collected);
434 result.push_back(base64_chars[buffer & 0x3F]);
435 }
436 while (result.size() % 4)
437 result.push_back('=');
438 return result;
439 }

448 {

450 "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/";
451 std::string result;
452 result.reserve(input.size() / 4 * 3);
453 uint32_t buffer = 0;
454 int bits_collected = 0;
455 for (char c : input)
456 {
457 if (c == '=')
458 break;
459 size_t index = base64_chars.find(c);
460 if (index == std::string::npos)
461 continue;
462 buffer = (buffer << 6) | index;
463 bits_collected += 6;
464 if (bits_collected >= 8)
465 {
466 bits_collected -= 8;
467 result.push_back((buffer >> bits_collected) & 0xFF);
468 }
469 }
470 return result;

631 state[idx + 2] ^ gmul(state[idx + 3], 2);

674 inline void key_expansion(const uint8_t *key, uint8_t *round_keys)
675 {
676 uint8_t temp[4];
677 uint8_t rcon[4];
678 for (int i = 0; i < 16; i++)
679 round_keys[i] = key[i];
680 for (int i = 4; i < 44; i++)
681 {
682 std::memcpy(temp, &round_keys[(i - 1) * 4], 4);
683 if (i % 4 == 0)
684 {
685 uint8_t t = temp[0];
686 temp[0] = Sbox[temp[1]];
687 temp[1] = Sbox[temp[2]];
688 temp[2] = Sbox[temp[3]];
689 temp[3] = Sbox[t];
690 rcon[0] = 0x01 << ((i / 4) - 1);
691 rcon[1] = rcon[2] = rcon[3] = 0;
692 if (rcon[0] == 0)
693 rcon[0] = 0x1b;
694 for (int j = 0; j < 4; j++)
695 temp[j] ^= rcon[j];
696 }
697 for (int j = 0; j < 4; j++)
698 {
699 round_keys[i * 4 + j] = round_keys[(i - 4) * 4 + j] ^ temp[j];
700 }
701 }
