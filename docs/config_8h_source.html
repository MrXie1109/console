119 void read_from_stream(std::istream &is) {
120 std::string line;
121 string_type current_section;
122 while (std::getline(is, line)) {
123 string_type trimmed_line
124 = trim(string_type(line.begin(), line.end()));
125 if (trimmed_line.empty() || trimmed_line[0] == CharT(';')
126 || trimmed_line[0] == CharT('#'))
127 continue;
128 if (trimmed_line.front() == CharT('[')
129 && trimmed_line.back() == CharT(']'))
130 current_section
131 = trim(trimmed_line.substr(1, trimmed_line.size() - 2));
132 else {
133 auto pos = trimmed_line.find(CharT('='));
134 if (pos != string_type::npos) {
135 string_type key = trim(trimmed_line.substr(0, pos));
136 string_type value = trim(trimmed_line.substr(pos + 1));
137 data_[current_section][key] = value;
138 }
139 }
140 }

244 Item get(const string_type &section_and_key) const {
245 auto pr = partition(section_and_key, string_type(1, CharT('.')));
246 if (pr.middle.empty())
247 throw IndexError("Invalid Section and Key Format: \""
248 + to_narrow(section_and_key) + '"');
249 auto section = pr.left;
250 auto key = pr.right;
251 auto sec_it = data_.find(section);
252 if (sec_it == data_.end())
253 throw IndexError(
254 "Section \"" + to_narrow(section) + "\" Not Found");
255 auto key_it = sec_it->second.find(key);
256 if (key_it == sec_it->second.end())
257 throw IndexError("Key \"" + to_narrow(key)
258 + "\" Not Found in Section \""
259 + to_narrow(section) + '"');
260 return Item(key_it->second);
261 }

306 bool has(const string_type &section_and_key) const {
307 auto pr = partition(section_and_key, string_type(1, CharT('.')));
308 if (pr.middle.empty()) {
309 auto section = pr.left;
310 return data_.find(section) != data_.end();
311 } else {
312 auto section = pr.left;
313 auto key = pr.right;
314 auto sec_it = data_.find(section);
315 if (sec_it == data_.end()) return false;
316 return sec_it->second.find(key) != sec_it->second.end();
317 }
318 }

325 bool remove(const string_type &section_and_key) {
326 auto pr = partition(section_and_key, string_type(1, CharT('.')));
327 if (pr.middle.empty()) {
328 auto section = pr.left;
329 return data_.erase(section) > 0;
330 } else {
331 auto section = pr.left;
332 auto key = pr.right;
333 auto sec_it = data_.find(section);
334 if (sec_it == data_.end()) return false;
335 return sec_it->second.erase(key) > 0;
