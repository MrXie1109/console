118 friend std::istream &operator>>(std::istream &is, INIConfig &config) {
119 std::string line;
120 std::string current_section;
121 while (std::getline(is, line)) {
122 std::string trimmed_line = trim(line);
123 if (trimmed_line.empty() || trimmed_line[0] == ';'
124 || trimmed_line[0] == '#')
125 continue;
126 if (trimmed_line.front() == '[' && trimmed_line.back() == ']')
127 current_section
128 = trim(trimmed_line.substr(1, trimmed_line.size() - 2));
129 else {
130 auto pos = trimmed_line.find('=');
131 if (pos != std::string::npos) {
132 std::string key = trim(trimmed_line.substr(0, pos));
133 std::string value = trim(trimmed_line.substr(pos + 1));
134 config.data_[current_section][key] = value;
135 }
136 }
137 }
138 return is;

197 Item get(const std::string &section_and_key) const {
198 auto pr = partition(section_and_key, ".");
199 if (pr.middle.empty())
200 throw IndexError("Invalid Section and Key Format: \""
